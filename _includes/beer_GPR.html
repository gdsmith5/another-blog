4.3 Run GPR with all variables using 5-fold CV

% opts = struct('UseParallel',true,'Kfold',5,'MaxObjectiveEvaluations',30,'Repartition',true,'Verbose',0);
k = 5;
c = cvpartition(length(extract_data),'KFold',k);

RMSECV = [];
iter = 25;
for j = 1:iter
    c = repartition(c);
    for i = 1:k
        train_idx = training(c,i);
        test_idx = ~train_idx;
        X = beer_spectra(train_idx,:); y = extract_data(train_idx);
        gprMdl = fitrgp(X,y);
        y_pred = predict(gprMdl,beer_spectra(test_idx,:));
        RMSECV(j,i) = sqrt(sumsqr(y_pred-extract_data(test_idx))/numel(y_pred));
    end
end
RMSECV = mean(mean(RMSECV))

% opts = struct('UseParallel',true,'Kfold',5,'MaxObjectiveEvaluations',30,'Repartition',true,'Verbose',0);
k = 5;

iter = 25;
for j = 1:iter
    c = repartition(c);
    for i = 1:k
        train_idx = training(c,i);
        test_idx = ~train_idx;
        X = beer_spectra(train_idx,:); y = extract_data(train_idx);
        [idx,weights] = relieff(X,y,5);
        idx = idx(1:round(m/10));
        gprMdl = fitrgp(X(:,idx),y);
        y_pred = predict(gprMdl,beer_spectra(test_idx,idx));
        RMSECV(j,i) = sqrt(sumsqr(y_pred-extract_data(test_idx))/numel(y_pred));
    end
end
RMSECV = mean(mean(RMSECV));
